132 void Periodic() override {
133 frc::SmartDashboard::PutBoolean(m_name + " Pid Enabled", m_pidEnabled);
134 frc::SmartDashboard::PutNumber(m_name + " Position",
135 GetCurrentPosition().value());
136 if (m_config.conversionFunction) {
137 std::string curPos =
138 m_config.conversionFunction.value()(GetCurrentPosition());
139 frc::SmartDashboard::PutString(m_name + " Converted Position", curPos);
140 }
141
142 if (m_controller.GetAbsoluteEncoderPosition().has_value())
143 frc::SmartDashboard::PutNumber(
144 m_name + " Absolute Position",
145 m_controller.GetAbsoluteEncoderPosition().value());
146
147 if (m_controller.GetAbsoluteEncoderPosition().has_value()) {
148 Distance_t absEncValue = Distance_t(
149 std::abs(m_controller.GetAbsoluteEncoderPosition().value()));
150
151 if (!resetOccurred && absEncValue <= m_config.tolerance) {
152 m_controller.ResetEncoder();
153 resetOccurred = true;
154 } else if (resetOccurred && absEncValue > m_config.tolerance) {
155 resetOccurred = false;
156 }
157 }
158
159 if (m_pidEnabled) {
160 frc2::TrapezoidProfileSubsystem<TDistance>::Periodic();
161 m_controller.Update();
162 }
163
164 if (!m_pidEnabled && !IsMovementAllowed(m_latestSpeed)) {
165 ConsoleWriter.logInfo(m_name,
166 "Periodic: Movement with speed %f is not allowed",
167 m_latestSpeed);
168
169 Stop();
170 }
